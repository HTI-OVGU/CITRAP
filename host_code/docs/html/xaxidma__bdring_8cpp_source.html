70 /******************************************************************************
71  * Compute the physical address of a descriptor from its virtual address
72  *
73  * @param BdPtr is the virtual address of the BD
74  *
75  * @returns Physical address of BdPtr
76  *
77  * @note Assume virtual and physical mapping is flat.
78  * RingPtr is an implicit parameter
79  *
80  *****************************************************************************/
81 #define XAXIDMA_VIRT_TO_PHYS(BdPtr) \
82  ((UINTPTR)(BdPtr) + (RingPtr->FirstBdPhysAddr - RingPtr->FirstBdAddr))
83 
84 /******************************************************************************
85  * Move the BdPtr argument ahead an arbitrary number of BDs wrapping around
86  * to the beginning of the ring if needed.
87  *
88  * We know if a wraparound should occur if the new BdPtr is greater than
89  * the high address in the ring OR if the new BdPtr crosses the 0xFFFFFFFF
90  * to 0 boundary.
91  *
92  * @param RingPtr is the ring BdPtr appears in
93  * @param BdPtr on input is the starting BD position and on output is the
94  * final BD position
95  * @param NumBd is the number of BD spaces to increment
96  *
97  * @returns None
98  *
99  * @note This function can be used only when DMA is in SG mode
100  *
101  *****************************************************************************/
102 #define XAXIDMA_RING_SEEKAHEAD(RingPtr, BdPtr, NumBd) \
103  { \
104  UINTPTR Addr = (UINTPTR)(void *)(BdPtr); \
105  \
106  Addr += ((RingPtr)->Separation * (NumBd)); \
107  if ((Addr > (RingPtr)->LastBdAddr) || ((UINTPTR)(BdPtr) > Addr)) \
108  { \
109  Addr -= (RingPtr)->Length; \
110  } \
111  \
112  (BdPtr) = (XAxiDma_Bd*)(void *)Addr; \
113  }
114 /******************************************************************************
115  * Move the BdPtr argument backwards an arbitrary number of BDs wrapping
116  * around to the end of the ring if needed.
117  *
118  * We know if a wraparound should occur if the new BdPtr is less than
119  * the base address in the ring OR if the new BdPtr crosses the 0xFFFFFFFF
120  * to 0 boundary.
121  *
122  * @param RingPtr is the ring BdPtr appears in
123  * @param BdPtr on input is the starting BD position and on output is the
124  * final BD position
125  * @param NumBd is the number of BD spaces to increment
126  *
127  * @returns None
128  *
129  * @note This function can be used only when DMA is in SG mode
130  *
131  *****************************************************************************/
132 #define XAXIDMA_RING_SEEKBACK(RingPtr, BdPtr, NumBd) \
133  { \
134  UINTPTR Addr = (UINTPTR)(BdPtr); \
135  \
136  Addr -= ((RingPtr)->Separation * (NumBd)); \
137  if ((Addr < (RingPtr)->FirstBdAddr) || ((UINTPTR)(BdPtr) < Addr)) \
138  { \
139  Addr += (RingPtr)->Length; \
140  } \
141  \
142  (BdPtr) = (XAxiDma_Bd*)Addr; \
143  }

164 int XAxiDma_UpdateBdRingCDesc(XAxiDma_BdRing *RingPtr)
165 {
166  UINTPTR RegBase;
167  UINTPTR BdPtr;
168  int RingIndex = RingPtr->RingIndex;
169 
170  /* BD list has yet to be created for this channel */
171  if (RingPtr->AllCnt == 0) {
172 
173  xdbg_printf(XDBG_DEBUG_ERROR, "BdRingStart: no bds\r\n");
174 
175  return XST_DMA_SG_NO_LIST;
176  }
177 
178  /* Do nothing if already started */
179  if (RingPtr->RunState == AXIDMA_CHANNEL_NOT_HALTED) {
180  /* Need to update tail pointer if needed (Engine is not
181  * transferring)
182  */
183  return XST_SUCCESS;
184  }
185 
186  if (!XAxiDma_BdRingHwIsStarted(RingPtr)) {
187  /* If hardware is not running, then we need to put a valid current
188  * BD pointer to the current BD register before start the hardware
189  */
190  RegBase = RingPtr->ChanBase;
191 
192  /* Put a valid BD pointer in the current BD pointer register
193  * So, the hardware is ready to go when tail BD pointer is updated
194  */
195  BdPtr = (UINTPTR)(void *)(RingPtr->BdaRestart);
196 
197  if (!XAxiDma_BdHwCompleted(BdPtr)) {
198  if (RingPtr->IsRxChannel) {
199  if (!RingIndex) {
200  XAxiDma_WriteReg(RegBase,
201  XAXIDMA_CDESC_OFFSET,
202  (XAXIDMA_VIRT_TO_PHYS(BdPtr) & XAXIDMA_DESC_LSB_MASK));
203  if (RingPtr->Addr_ext)
204  XAxiDma_WriteReg(RegBase,
205  XAXIDMA_CDESC_MSB_OFFSET,
206  UPPER_32_BITS(XAXIDMA_VIRT_TO_PHYS(BdPtr)));
207  } else {
208  XAxiDma_WriteReg(RegBase,
209  (XAXIDMA_RX_CDESC0_OFFSET +
210  (RingIndex - 1) * XAXIDMA_RX_NDESC_OFFSET),
211  (XAXIDMA_VIRT_TO_PHYS(BdPtr) & XAXIDMA_DESC_LSB_MASK));
212  if (RingPtr->Addr_ext)
213  XAxiDma_WriteReg(RegBase,
214  (XAXIDMA_RX_CDESC0_MSB_OFFSET +
215  (RingIndex - 1) * XAXIDMA_RX_NDESC_OFFSET),
216  UPPER_32_BITS(XAXIDMA_VIRT_TO_PHYS(BdPtr)));
217  }
218  } else {
219  XAxiDma_WriteReg(RegBase,
220  XAXIDMA_CDESC_OFFSET,
221  (XAXIDMA_VIRT_TO_PHYS(BdPtr) & XAXIDMA_DESC_LSB_MASK));
222  if (RingPtr->Addr_ext)
223  XAxiDma_WriteReg(RegBase, XAXIDMA_CDESC_MSB_OFFSET,
224  UPPER_32_BITS(XAXIDMA_VIRT_TO_PHYS(BdPtr)));
225  }
226  } else {
227  /* Look for an uncompleted BD
228  */
229  while (XAxiDma_BdHwCompleted(BdPtr)) {
230  BdPtr = XAxiDma_BdRingNext(RingPtr, BdPtr);
231 
232  if ((UINTPTR)BdPtr == (UINTPTR) RingPtr->BdaRestart) {
233  xdbg_printf(XDBG_DEBUG_ERROR,
234  "StartBdRingHw: Cannot find valid cdesc\r\n");
235 
236  return XST_DMA_ERROR;
237  }
238 
239  if (!XAxiDma_BdHwCompleted(BdPtr)) {
240  if (RingPtr->IsRxChannel) {
241  if (!RingIndex) {
242  XAxiDma_WriteReg(RegBase,
243  XAXIDMA_CDESC_OFFSET,
244  (XAXIDMA_VIRT_TO_PHYS(BdPtr) & XAXIDMA_DESC_LSB_MASK));
245  if (RingPtr->Addr_ext)
246  XAxiDma_WriteReg(RegBase, XAXIDMA_CDESC_MSB_OFFSET,
247  UPPER_32_BITS(XAXIDMA_VIRT_TO_PHYS(BdPtr)));
248  } else {
249  XAxiDma_WriteReg(RegBase,
250  (XAXIDMA_RX_CDESC0_OFFSET +
251  (RingIndex - 1) * XAXIDMA_RX_NDESC_OFFSET),
252  (XAXIDMA_VIRT_TO_PHYS(BdPtr) & XAXIDMA_DESC_LSB_MASK));
253  if (RingPtr->Addr_ext)
254  XAxiDma_WriteReg(RegBase,
255  (XAXIDMA_RX_CDESC0_MSB_OFFSET +
256  (RingIndex - 1) * XAXIDMA_RX_NDESC_OFFSET),
257  UPPER_32_BITS(XAXIDMA_VIRT_TO_PHYS(BdPtr)));
258  }
259  } else {
260  XAxiDma_WriteReg(RegBase,
261  XAXIDMA_CDESC_OFFSET,
262  (XAXIDMA_VIRT_TO_PHYS(BdPtr) & XAXIDMA_DESC_LSB_MASK));
263  if (RingPtr->Addr_ext)
264  XAxiDma_WriteReg(RegBase, XAXIDMA_CDESC_MSB_OFFSET,
265  UPPER_32_BITS(XAXIDMA_VIRT_TO_PHYS(BdPtr)));
266  }
267  break;
268  }
269  }
270  }
271 
272  }
273 
274  return XST_SUCCESS;
275 }

317 u32 XAxiDma_BdRingCreate(XAxiDma_BdRing *RingPtr, UINTPTR PhysAddr,
318  UINTPTR VirtAddr, u32 Alignment, int BdCount)
319 {
320  int i;
321  UINTPTR BdVirtAddr;
322  UINTPTR BdPhysAddr;
323 
324  if (BdCount <= 0) {
325 
326  xdbg_printf(XDBG_DEBUG_ERROR, "BdRingCreate: non-positive BD"
327  " number %d\r\n", BdCount);
328 
329  return XST_INVALID_PARAM;
330  }
331 
332  /* In case there is a failure prior to creating list, make sure the
333  * following attributes are 0 to prevent calls to other SG functions
334  * from doing anything
335  */
336  RingPtr->AllCnt = 0;
337  RingPtr->FreeCnt = 0;
338  RingPtr->HwCnt = 0;
339  RingPtr->PreCnt = 0;
340  RingPtr->PostCnt = 0;
341  RingPtr->Cyclic = 0;
342 
343  /* Make sure Alignment parameter meets minimum requirements */
344  if (Alignment < XAXIDMA_BD_MINIMUM_ALIGNMENT) {
345 
346  xdbg_printf(XDBG_DEBUG_ERROR, "BdRingCreate: alignment too "
347  "small %d, need to be at least %d\r\n", (int)Alignment,
348  XAXIDMA_BD_MINIMUM_ALIGNMENT);
349 
350  return XST_INVALID_PARAM;
351  }
352 
353  /* Make sure Alignment is a power of 2 */
354  if ((Alignment - 1) & Alignment) {
355 
356  xdbg_printf(XDBG_DEBUG_ERROR, "BdRingCreate: alignment not"
357  " valid %d\r\n", (int)Alignment);
358 
359  return XST_INVALID_PARAM;
360  }
361 
362  /* Make sure PhysAddr and VirtAddr are on same Alignment */
363  if ((PhysAddr % Alignment) || (VirtAddr % Alignment)) {
364 
365  xdbg_printf(XDBG_DEBUG_ERROR, "BdRingCreate: Physical address"
366  " %x and virtual address %x have different alignment\r\n",
367  (unsigned int)PhysAddr, (unsigned int)VirtAddr);
368 
369  return XST_INVALID_PARAM;
370  }
371 
372  /* Compute how many bytes will be between the start of adjacent BDs */
373  RingPtr->Separation =
374  (sizeof(XAxiDma_Bd) + (Alignment - 1)) & ~(Alignment - 1);
375 
376  /* Must make sure the ring doesn't span address 0x00000000. If it does,
377  * then the next/prev BD traversal macros will fail.
378  */
379  if (VirtAddr > (VirtAddr + (RingPtr->Separation * BdCount) - 1)) {
380 
381  xdbg_printf(XDBG_DEBUG_ERROR, "BdRingCreate: BD space cross "
382  "0x0\r\n");
383 
384  return XST_DMA_SG_LIST_ERROR;
385  }
386 
387  /* Initial ring setup:
388  * - Clear the entire space
389  * - Setup each BD's next pointer with the physical address of the
390  * next BD
391  * - Put hardware information in each BD
392  */
393  memset((void *) VirtAddr, 0, (RingPtr->Separation * BdCount));
394 
395  BdVirtAddr = VirtAddr;
396  BdPhysAddr = PhysAddr + RingPtr->Separation;
397  for (i = 1; i < BdCount; i++) {
398  XAxiDma_BdWrite(BdVirtAddr, XAXIDMA_BD_NDESC_OFFSET,
399  (BdPhysAddr & XAXIDMA_DESC_LSB_MASK));
400  XAxiDma_BdWrite(BdVirtAddr, XAXIDMA_BD_NDESC_MSB_OFFSET,
401  UPPER_32_BITS(BdPhysAddr));
402 
403  /* Put hardware information in the BDs
404  */
405  XAxiDma_BdWrite(BdVirtAddr, XAXIDMA_BD_HAS_STSCNTRL_OFFSET,
406  (u32)RingPtr->HasStsCntrlStrm);
407 
408  XAxiDma_BdWrite(BdVirtAddr, XAXIDMA_BD_HAS_DRE_OFFSET,
409  (((u32)(RingPtr->HasDRE)) << XAXIDMA_BD_HAS_DRE_SHIFT) |
410  RingPtr->DataWidth);
411 
412  // XAXIDMA_CACHE_FLUSH(BdVirtAddr);
413  BdVirtAddr += RingPtr->Separation;
414  BdPhysAddr += RingPtr->Separation;
415  }
416 
417  /* At the end of the ring, link the last BD back to the top */
418  XAxiDma_BdWrite(BdVirtAddr, XAXIDMA_BD_NDESC_OFFSET,
419  (PhysAddr & XAXIDMA_DESC_LSB_MASK));
420  XAxiDma_BdWrite(BdVirtAddr, XAXIDMA_BD_NDESC_MSB_OFFSET,
421  UPPER_32_BITS(PhysAddr));
422 
423 
424  /* Setup the last BD's hardware information */
425  XAxiDma_BdWrite(BdVirtAddr, XAXIDMA_BD_HAS_STSCNTRL_OFFSET,
426  (u32)RingPtr->HasStsCntrlStrm);
427 
428  XAxiDma_BdWrite(BdVirtAddr, XAXIDMA_BD_HAS_DRE_OFFSET,
429  (((u32)(RingPtr->HasDRE)) << XAXIDMA_BD_HAS_DRE_SHIFT) |
430  RingPtr->DataWidth);
431 
432  /* Setup and initialize pointers and counters */
433  RingPtr->RunState = AXIDMA_CHANNEL_HALTED;
434  RingPtr->FirstBdAddr = VirtAddr;
435  RingPtr->FirstBdPhysAddr = PhysAddr;
436  RingPtr->LastBdAddr = BdVirtAddr;
437  RingPtr->Length = RingPtr->LastBdAddr - RingPtr->FirstBdAddr +
438  RingPtr->Separation;
439  RingPtr->AllCnt = BdCount;
440  RingPtr->FreeCnt = BdCount;
441  RingPtr->FreeHead = (XAxiDma_Bd *) VirtAddr;
442  RingPtr->PreHead = (XAxiDma_Bd *) VirtAddr;
443  RingPtr->HwHead = (XAxiDma_Bd *) VirtAddr;
444  RingPtr->HwTail = (XAxiDma_Bd *) VirtAddr;
445  RingPtr->PostHead = (XAxiDma_Bd *) VirtAddr;
446  RingPtr->BdaRestart = (XAxiDma_Bd *) VirtAddr;
447  RingPtr->CyclicBd = (XAxiDma_Bd *) malloc(sizeof(XAxiDma_Bd));
448 
449  return XST_SUCCESS;
450 }

476 int XAxiDma_BdRingClone(XAxiDma_BdRing *RingPtr, XAxiDma_Bd *SrcBdPtr)
477 {
478  int i;
479  UINTPTR CurBd;
480  u32 Save;
481  XAxiDma_Bd TmpBd;
482 
483  /* Can't do this function if there isn't a ring */
484  if (RingPtr->AllCnt == 0) {
485 
486  xdbg_printf(XDBG_DEBUG_ERROR, "BdRingClone: no bds\r\n");
487 
488  return XST_DMA_SG_NO_LIST;
489  }
490 
491  /* Can't do this function with the channel running */
492  if (RingPtr->RunState == AXIDMA_CHANNEL_NOT_HALTED) {
493 
494  xdbg_printf(XDBG_DEBUG_ERROR, "BdRingClone: bd ring started "
495  "already, cannot do\r\n");
496 
497  return XST_DEVICE_IS_STARTED;
498  }
499 
500  /* Can't do this function with some of the BDs in use */
501  if (RingPtr->FreeCnt != RingPtr->AllCnt) {
502 
503  xdbg_printf(XDBG_DEBUG_ERROR, "BdRingClone: some bds already "
504  "in use %d/%d\r\n", RingPtr->FreeCnt, RingPtr->AllCnt);
505 
506  return XST_DMA_SG_LIST_ERROR;
507  }
508 
509  /* Make a copy of the template then modify it by clearing
510  * the complete bit in status/control field
511  */
512  memcpy(&TmpBd, SrcBdPtr, sizeof(XAxiDma_Bd));
513 
514  Save = XAxiDma_BdRead(&TmpBd, XAXIDMA_BD_STS_OFFSET);
515  Save &= ~XAXIDMA_BD_STS_COMPLETE_MASK;
516  XAxiDma_BdWrite(&TmpBd, XAXIDMA_BD_STS_OFFSET, Save);
517 
518  for (i = 0, CurBd = RingPtr->FirstBdAddr;
519  i < RingPtr->AllCnt; i++, CurBd += RingPtr->Separation) {
520 
521  memcpy((void *)((UINTPTR)CurBd + XAXIDMA_BD_START_CLEAR),
522  (void *)((UINTPTR)(&TmpBd) + XAXIDMA_BD_START_CLEAR),
523  XAXIDMA_BD_BYTES_TO_CLEAR);
524 
525  // XAXIDMA_CACHE_FLUSH(CurBd);
526  }
527 
528  return XST_SUCCESS;
529 }

550 int XAxiDma_StartBdRingHw(XAxiDma_BdRing *RingPtr)
551 {
552  UINTPTR RegBase;
553  int RingIndex = RingPtr->RingIndex;
554 
555  if (!XAxiDma_BdRingHwIsStarted(RingPtr)) {
556  /* Start the hardware
557  */
558  RegBase = RingPtr->ChanBase;
559  XAxiDma_WriteReg(RegBase, XAXIDMA_CR_OFFSET,
560  XAxiDma_ReadReg(RegBase, XAXIDMA_CR_OFFSET)
561  | XAXIDMA_CR_RUNSTOP_MASK);
562  }
563 
564  if (XAxiDma_BdRingHwIsStarted(RingPtr)) {
565  /* Note as started */
566  RingPtr->RunState = AXIDMA_CHANNEL_NOT_HALTED;
567 
568  /* If there are unprocessed BDs then we want the channel to begin
569  * processing right away
570  */
571  if (RingPtr->HwCnt > 0) {
572 
573  // XAXIDMA_CACHE_INVALIDATE(RingPtr->HwTail);
574  if (RingPtr->Cyclic) {
575  XAxiDma_WriteReg(RingPtr->ChanBase,
576  XAXIDMA_TDESC_OFFSET,
577  (u32)XAXIDMA_VIRT_TO_PHYS(RingPtr->CyclicBd));
578  if (RingPtr->Addr_ext)
579  XAxiDma_WriteReg(RingPtr->ChanBase,
580  XAXIDMA_TDESC_MSB_OFFSET,
581  UPPER_32_BITS(XAXIDMA_VIRT_TO_PHYS(RingPtr->CyclicBd)));
582  return XST_SUCCESS;
583  }
584 
585  if ((XAxiDma_BdRead(RingPtr->HwTail,
586  XAXIDMA_BD_STS_OFFSET) &
587  XAXIDMA_BD_STS_COMPLETE_MASK) == 0) {
588  if (RingPtr->IsRxChannel) {
589  if (!RingIndex) {
590  XAxiDma_WriteReg(RingPtr->ChanBase,
591  XAXIDMA_TDESC_OFFSET, (XAXIDMA_VIRT_TO_PHYS(RingPtr->HwTail) & XAXIDMA_DESC_LSB_MASK));
592  if (RingPtr->Addr_ext)
593  XAxiDma_WriteReg(RingPtr->ChanBase, XAXIDMA_TDESC_MSB_OFFSET,
594  UPPER_32_BITS(XAXIDMA_VIRT_TO_PHYS(RingPtr->HwTail)));
595  } else {
596  XAxiDma_WriteReg(RingPtr->ChanBase,
597  (XAXIDMA_RX_TDESC0_OFFSET +
598  (RingIndex - 1) * XAXIDMA_RX_NDESC_OFFSET),
599  (XAXIDMA_VIRT_TO_PHYS(RingPtr->HwTail) & XAXIDMA_DESC_LSB_MASK ));
600  if (RingPtr->Addr_ext)
601  XAxiDma_WriteReg(RingPtr->ChanBase,
602  (XAXIDMA_RX_TDESC0_MSB_OFFSET +
603  (RingIndex - 1) * XAXIDMA_RX_NDESC_OFFSET),
604  UPPER_32_BITS(XAXIDMA_VIRT_TO_PHYS(RingPtr->HwTail)));
605  }
606  } else {
607  XAxiDma_WriteReg(RingPtr->ChanBase,
608  XAXIDMA_TDESC_OFFSET, (XAXIDMA_VIRT_TO_PHYS(RingPtr->HwTail) & XAXIDMA_DESC_LSB_MASK));
609  if (RingPtr->Addr_ext)
610  XAxiDma_WriteReg(RingPtr->ChanBase, XAXIDMA_TDESC_MSB_OFFSET,
611  UPPER_32_BITS(XAXIDMA_VIRT_TO_PHYS(RingPtr->HwTail)));
612  }
613  }
614  }
615 
616  return XST_SUCCESS;
617  }
618 
619  return XST_DMA_ERROR;
620 }

685 int XAxiDma_BdRingSetCoalesce(XAxiDma_BdRing *RingPtr, u32 Counter, u32 Timer)
686 {
687  u32 Cr;
688 
689  Cr = XAxiDma_ReadReg(RingPtr->ChanBase, XAXIDMA_CR_OFFSET);
690 
691  if (Counter != XAXIDMA_NO_CHANGE) {
692  if ((Counter == 0) || (Counter > 0xFF)) {
693 
694  xdbg_printf(XDBG_DEBUG_ERROR, "BdRingSetCoalesce: "
695  "invalid coalescing threshold %d", (int)Counter);
696  return XST_FAILURE;
697  }
698 
699  Cr = (Cr & ~XAXIDMA_COALESCE_MASK) |
700  (Counter << XAXIDMA_COALESCE_SHIFT);
701  }
702 
703  if (Timer != XAXIDMA_NO_CHANGE) {
704  if (Timer > 0xFF) {
705 
706  xdbg_printf(XDBG_DEBUG_ERROR, "BdRingSetCoalesce: "
707  "invalid delay counter %d", (int)Timer);
708 
709  return XST_FAILURE;
710  }
711 
712  Cr = (Cr & ~XAXIDMA_DELAY_MASK) |
713  (Timer << XAXIDMA_DELAY_SHIFT);
714  }
715 
716  XAxiDma_WriteReg(RingPtr->ChanBase, XAXIDMA_CR_OFFSET, Cr);
717 
718  return XST_SUCCESS;
719 }

917 int XAxiDma_BdRingUnAlloc(XAxiDma_BdRing *RingPtr, int NumBd,

1213 int XAxiDma_BdRingFromHw(XAxiDma_BdRing *RingPtr, int BdLimit,
1214  XAxiDma_Bd **BdSetPtr)
1215 {

1218  int BdPartialCount;
1219  u32 BdSts;
1220  u32 BdCr;
1221

1223  BdCount = 0;
1224  BdPartialCount = 0;
1225  BdSts = 0;
1226  BdCr = 0;
1227

1229  if (RingPtr->HwCnt == 0) {

1233  }
1234

1236  BdLimit = RingPtr->HwCnt;
1237  }
1238 
1239  /* Starting at HwHead, keep moving forward in the list until:
1240  * - A BD is encountered with its completed bit clear in the status
1241  * word which means hardware has not completed processing of that
1242  * BD.
1243  * - RingPtr->HwTail is reached
1244  * - The number of requested BDs has been processed

1249  // XAXIDMA_CACHE_INVALIDATE(CurBdPtr);

1258  }

1263  * it is clear, then there are more BDs for the current packet.

1270  (BdCr & XAXIDMA_BD_CTRL_TXEOF_MASK)) ||
1271  ((RingPtr->IsRxChannel) && (BdSts &
1272  XAXIDMA_BD_STS_RXEOF_MASK)))) {

1282  // XAXIDMA_CACHE_FLUSH(CurBdPtr);
1283  }
1284

1286  if (CurBdPtr == RingPtr->HwTail) {

1288  }
1289

1291  CurBdPtr = (XAxiDma_Bd *)((void *)XAxiDma_BdRingNext(RingPtr, CurBdPtr));
1292  }
1293 
1294  /* Subtract off any partial packet BDs found */
1295  BdCount -= BdPartialCount;
1296 
1297  /* If BdCount is non-zero then BDs were found to return. Set return
1298  * parameters, update pointers and counters, return success
1299  */
1300  if (BdCount) {

1304  RingPtr->PostCnt += BdCount;

1307 
1308  return BdCount;
1309  } else {
1310  *BdSetPtr = (XAxiDma_Bd *)NULL;
1311 
1312  return 0;
1313  }
