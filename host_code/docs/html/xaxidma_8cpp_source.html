114 int XAxiDma_CfgInitialize(XAxiDma *InstancePtr, XAxiDma_Config *Config)
115 {
116  printf("\r\n--- Entering XAxiDma_CfgInitialize() --- \r\n");
117 
118  UINTPTR BaseAddr;
119  int TimeOut;
120  int Index;
121  u32 MaxTransferLen;
122 
123  InstancePtr->Initialized = 0;
124 
125  if (!Config) {
126  return XST_INVALID_PARAM;
127  }
128 
129  BaseAddr = Config->BaseAddr;
130 
131  /* Setup the instance */
132  memset(InstancePtr, 0, sizeof(XAxiDma));
133  InstancePtr->RegBase = BaseAddr;
134 
135  /* Get hardware setting information from the configuration structure
136  */
137  InstancePtr->HasMm2S = Config->HasMm2S;
138  InstancePtr->HasS2Mm = Config->HasS2Mm;
139 
140  InstancePtr->HasSg = Config->HasSg;
141 
142  InstancePtr->MicroDmaMode = Config->MicroDmaMode;
143  InstancePtr->AddrWidth = Config->AddrWidth;
144 
145  /* Get the number of channels */
146  InstancePtr->TxNumChannels = Config->Mm2sNumChannels;
147  InstancePtr->RxNumChannels = Config->S2MmNumChannels;
148 
149  /* This condition is for IP version < 6.00a */
150  if (!InstancePtr->TxNumChannels) {
151  InstancePtr->TxNumChannels = 1;
152  }
153  if (!InstancePtr->RxNumChannels) {
154  InstancePtr->RxNumChannels = 1;
155  }
156 
157  if ((InstancePtr->RxNumChannels > 1) ||
158  (InstancePtr->TxNumChannels > 1)) {
159  MaxTransferLen =
160  XAXIDMA_MCHAN_MAX_TRANSFER_LEN;
161  } else {
162  MaxTransferLen = (1U << Config->SgLengthWidth) - 1;
163  }
164 
165  /* Initialize the ring structures */
166  InstancePtr->TxBdRing.RunState = AXIDMA_CHANNEL_HALTED;
167  InstancePtr->TxBdRing.IsRxChannel = 0;
168  if (!InstancePtr->MicroDmaMode) {
169  InstancePtr->TxBdRing.MaxTransferLen = MaxTransferLen;
170  } else {
171  /* In MicroDMA mode, Maximum length that can be transferred
172  * is '(Memory Data Width / 8) * Burst Size'
173  */
174  InstancePtr->TxBdRing.MaxTransferLen =
175  ((Config->Mm2SDataWidth / 8) *
176  Config->Mm2SBurstSize);
177  }
178  InstancePtr->TxBdRing.RingIndex = 0;
179 
180  for (Index = 0; Index < InstancePtr->RxNumChannels; Index++) {
181  InstancePtr->RxBdRing[Index].RunState
182  = AXIDMA_CHANNEL_HALTED;
183  InstancePtr->RxBdRing[Index].IsRxChannel = 1;
184  InstancePtr->RxBdRing[Index].RingIndex = Index;
185  }
186 
187  if (InstancePtr->HasMm2S) {
188  InstancePtr->TxBdRing.ChanBase =
189  BaseAddr + XAXIDMA_TX_OFFSET;
190  InstancePtr->TxBdRing.HasStsCntrlStrm =
191  Config->HasStsCntrlStrm;
192  if (InstancePtr->AddrWidth > 32) {
193  InstancePtr->TxBdRing.Addr_ext = 1;
194  } else {
195  InstancePtr->TxBdRing.Addr_ext = 0;
196  }
197 
198  InstancePtr->TxBdRing.HasDRE = Config->HasMm2SDRE;
199  InstancePtr->TxBdRing.DataWidth =
200  ((unsigned int)Config->Mm2SDataWidth >> 3);
201  }
202 
203  if (InstancePtr->HasS2Mm) {
204  for (Index = 0;
205  Index < InstancePtr->RxNumChannels; Index++) {
206  InstancePtr->RxBdRing[Index].ChanBase =
207  BaseAddr + XAXIDMA_RX_OFFSET;
208  InstancePtr->RxBdRing[Index].HasStsCntrlStrm =
209  Config->HasStsCntrlStrm;
210  InstancePtr->RxBdRing[Index].HasDRE =
211  Config->HasS2MmDRE;
212  InstancePtr->RxBdRing[Index].DataWidth =
213  ((unsigned int)Config->S2MmDataWidth >> 3);
214 
215  if (!InstancePtr->MicroDmaMode) {
216  InstancePtr->RxBdRing[Index].MaxTransferLen =
217  MaxTransferLen;
218  } else {
219  /* In MicroDMA mode, Maximum length that can be transferred
220  * is '(Memory Data Width / 8) * Burst Size'
221  */
222  InstancePtr->RxBdRing[Index].MaxTransferLen =
223  ((Config->S2MmDataWidth / 8) *
224  Config->S2MmBurstSize);
225  }
226  if (InstancePtr->AddrWidth > 32) {
227  InstancePtr->RxBdRing[Index].Addr_ext = 1;
228  } else {
229  InstancePtr->RxBdRing[Index].Addr_ext = 0;
230  }
231  }
232  }
233 
234  /* Reset the engine so the hardware starts from a known state
235  */
236  XAxiDma_Reset(InstancePtr);
237 
238  /* At the initialization time, hardware should finish reset quickly
239  */
240  TimeOut = XAXIDMA_RESET_TIMEOUT;
241 
242  while (TimeOut) {
243 
244  if (XAxiDma_ResetIsDone(InstancePtr)) {
245  break;
246  }
247 
248  TimeOut -= 1;
249 
250  }
251 
252  if (!TimeOut) {
253  xdbg_printf(XDBG_DEBUG_ERROR, "Failed reset in"
254  "initialize\r\n");
255 
256  /* Need system hard reset to recover
257  */
258  InstancePtr->Initialized = 0;
259  return XST_DMA_ERROR;
260  }
261 
262  /* Initialization is successful
263  */
264  InstancePtr->Initialized = 1;
265 
266  return XST_SUCCESS;
267 }

289 void XAxiDma_Reset(XAxiDma *InstancePtr)
290 {
291  printf("\r\n--- Entering XAxiDma_Reset --- \r\n");
292 
293  UINTPTR RegBase;
294  XAxiDma_BdRing *TxRingPtr;
295  XAxiDma_BdRing *RxRingPtr;
296  int RingIndex;
297 
298  if (InstancePtr->HasMm2S) {
299  TxRingPtr = XAxiDma_GetTxRing(InstancePtr);
300 
301  /* Save the locations of current BDs both rings are working on
302  * before the reset so later we can resume the rings smoothly.
303  */
304  if (XAxiDma_HasSg(InstancePtr)) {
305  XAxiDma_BdRingSnapShotCurrBd(TxRingPtr);
306 
307  for (RingIndex = 0; RingIndex < InstancePtr->RxNumChannels;
308  RingIndex++) {
309  RxRingPtr = XAxiDma_GetRxIndexRing(InstancePtr,
310  RingIndex);
311  XAxiDma_BdRingSnapShotCurrBd(RxRingPtr);
312  }
313  }
314 
315  }
316 
317  /* Reset
318  */
319  if (InstancePtr->HasMm2S) {
320  RegBase = InstancePtr->RegBase + XAXIDMA_TX_OFFSET;
321  } else {
322  RegBase = InstancePtr->RegBase + XAXIDMA_RX_OFFSET;
323  }
324 
325  XAxiDma_WriteReg(RegBase, XAXIDMA_CR_OFFSET, XAXIDMA_CR_RESET_MASK);
326 
327  /* Set TX/RX Channel state */
328  if (InstancePtr->HasMm2S) {
329  TxRingPtr = XAxiDma_GetTxRing(InstancePtr);
330 
331  TxRingPtr->RunState = AXIDMA_CHANNEL_HALTED;
332  }
333 
334  if (InstancePtr->HasS2Mm) {
335  for (RingIndex = 0; RingIndex < InstancePtr->RxNumChannels;
336  RingIndex++) {
337  RxRingPtr = XAxiDma_GetRxIndexRing(InstancePtr, RingIndex);
338  if (InstancePtr->HasS2Mm) {
339  RxRingPtr->RunState = AXIDMA_CHANNEL_HALTED;
340  }
341  }
342  }
343 
344 }

361 int XAxiDma_ResetIsDone(XAxiDma *InstancePtr)
362 {
363  u32 RegisterValue;
364  XAxiDma_BdRing *TxRingPtr;
365  XAxiDma_BdRing *RxRingPtr;
366 
367  TxRingPtr = XAxiDma_GetTxRing(InstancePtr);
368  RxRingPtr = XAxiDma_GetRxRing(InstancePtr);
369 
370  /* Check transmit channel
371  */
372  if (InstancePtr->HasMm2S) {
373  RegisterValue = XAxiDma_ReadReg(TxRingPtr->ChanBase,
374  XAXIDMA_CR_OFFSET);
375 
376  /* Reset is done when the reset bit is low
377  */
378  if (RegisterValue & XAXIDMA_CR_RESET_MASK) {
379 
380  return 0;
381  }
382  }
383 
384  /* Check receive channel
385  */
386  if (InstancePtr->HasS2Mm) {
387  RegisterValue = XAxiDma_ReadReg(RxRingPtr->ChanBase,
388  XAXIDMA_CR_OFFSET);
389 
390  /* Reset is done when the reset bit is low
391  */
392  if (RegisterValue & XAXIDMA_CR_RESET_MASK) {
393 
394  return 0;
395  }
396  }
397 
398  return 1;
399 }
400 /*****************************************************************************/
401 /*
402 * Start the DMA engine.
403 *
404 * Start a halted engine. Processing of BDs is not started.
405 *
406 * @param InstancePtr is a pointer to the DMA engine instance to be
407 * worked on.
408 *
409 * @return
410 * - XST_SUCCESS for success
411 * - XST_NOT_SGDMA if the driver instance has not been initialized
412 * - XST_DMA_ERROR if starting the hardware channel fails
413 *
414 * @note None
415 *
416 *****************************************************************************/
417 static int XAxiDma_Start(XAxiDma *InstancePtr)
418 {
419  int Status;
420 
421  if (!InstancePtr->Initialized) {
422 
423  xdbg_printf(XDBG_DEBUG_ERROR, "Start: Driver not initialized "
424  "%d\r\n", InstancePtr->Initialized);
425 
426  return XST_NOT_SGDMA;
427  }
428 
429  if (InstancePtr->HasMm2S) {
430  XAxiDma_BdRing *TxRingPtr;
431  TxRingPtr = XAxiDma_GetTxRing(InstancePtr);
432 
433  if (TxRingPtr->RunState == AXIDMA_CHANNEL_HALTED) {
434 
435  /* Start the channel
436  */
437  if (XAxiDma_HasSg(InstancePtr)) {
438  Status = XAxiDma_BdRingStart(TxRingPtr);
439  if (Status != XST_SUCCESS) {
440  xdbg_printf(XDBG_DEBUG_ERROR,
441  "Start hw tx channel failed %d\r\n",
442  Status);
443 
444  return XST_DMA_ERROR;
445  }
446  } else {
447  XAxiDma_WriteReg(TxRingPtr->ChanBase,
448  XAXIDMA_CR_OFFSET,
449  XAxiDma_ReadReg(TxRingPtr->ChanBase,
450  XAXIDMA_CR_OFFSET)
451  | XAXIDMA_CR_RUNSTOP_MASK);
452  }
453  TxRingPtr->RunState = AXIDMA_CHANNEL_NOT_HALTED;
454  }
455  }
456 
457  if (InstancePtr->HasS2Mm) {
458  int RingIndex = 0;
459 
460  for (RingIndex = 0; RingIndex < InstancePtr->RxNumChannels;
461  RingIndex++) {
462  XAxiDma_BdRing *RxRingPtr;
463  RxRingPtr = XAxiDma_GetRxIndexRing(InstancePtr,
464  RingIndex);
465 
466  if (RxRingPtr->RunState != AXIDMA_CHANNEL_HALTED) {
467  return XST_SUCCESS;
468  }
469 
470  /* Start the channel
471  */
472  if (XAxiDma_HasSg(InstancePtr)) {
473  Status = XAxiDma_BdRingStart(RxRingPtr);
474  if (Status != XST_SUCCESS) {
475  xdbg_printf(XDBG_DEBUG_ERROR,
476  "Start hw tx channel failed %d\r\n",
477  Status);
478 
479  return XST_DMA_ERROR;
480  }
481  } else {
482  XAxiDma_WriteReg(RxRingPtr->ChanBase,
483  XAXIDMA_CR_OFFSET,
484  XAxiDma_ReadReg(RxRingPtr->ChanBase,
485  XAXIDMA_CR_OFFSET) |
486  XAXIDMA_CR_RUNSTOP_MASK);
487  }
488 
489  RxRingPtr->RunState = AXIDMA_CHANNEL_NOT_HALTED;
490  }
491  }
492 
493  return XST_SUCCESS;
494 }

515 int XAxiDma_Pause(XAxiDma *InstancePtr)
516 {
517 
518  if (!InstancePtr->Initialized) {
519 
520  xdbg_printf(XDBG_DEBUG_ERROR, "Pause: Driver not initialized"
521  " %d\r\n", InstancePtr->Initialized);
522 
523  return XST_NOT_SGDMA;
524  }
525 
526  if (InstancePtr->HasMm2S) {
527  XAxiDma_BdRing *TxRingPtr;
528  TxRingPtr = XAxiDma_GetTxRing(InstancePtr);
529 
530  /* If channel is halted, then we do not need to do anything
531  */
532  if (!XAxiDma_HasSg(InstancePtr)) {
533  XAxiDma_WriteReg(TxRingPtr->ChanBase,
534  XAXIDMA_CR_OFFSET,
535  XAxiDma_ReadReg(TxRingPtr->ChanBase,
536  XAXIDMA_CR_OFFSET)
537  & ~XAXIDMA_CR_RUNSTOP_MASK);
538  }
539 
540  TxRingPtr->RunState = AXIDMA_CHANNEL_HALTED;
541  }
542 
543  if (InstancePtr->HasS2Mm) {
544  int RingIndex = 0;
545  for (RingIndex = 0; RingIndex < InstancePtr->RxNumChannels;
546  RingIndex++) {
547  XAxiDma_BdRing *RxRingPtr;
548  RxRingPtr = XAxiDma_GetRxIndexRing(InstancePtr, RingIndex);
549 
550  /* If channel is halted, then we do not need to do anything
551  */
552 
553  if (!XAxiDma_HasSg(InstancePtr) && !RingIndex) {
554  XAxiDma_WriteReg(RxRingPtr->ChanBase,
555  XAXIDMA_CR_OFFSET,
556  XAxiDma_ReadReg(RxRingPtr->ChanBase,
557  XAXIDMA_CR_OFFSET)
558  & ~XAXIDMA_CR_RUNSTOP_MASK);
559  }
560 
561  RxRingPtr->RunState = AXIDMA_CHANNEL_HALTED;
562  }
563  }
564 
565  return XST_SUCCESS;
566 
567 }

584 int XAxiDma_Resume(XAxiDma *InstancePtr)
585 {
586  int Status;
587 
588  if (!InstancePtr->Initialized) {
589 
590  xdbg_printf(XDBG_DEBUG_ERROR, "Resume: Driver not initialized"
591  " %d\r\n", InstancePtr->Initialized);
592 
593  return XST_NOT_SGDMA;
594  }
595 
596  /* If the DMA engine is not running, start it. Start may fail.
597  */
598  if (!XAxiDma_Started(InstancePtr)) {
599  Status = XAxiDma_Start(InstancePtr);
600 
601  if (Status != XST_SUCCESS) {
602  xdbg_printf(XDBG_DEBUG_ERROR, "Resume: failed to start"
603  " engine %d\r\n", Status);
604 
605  return Status;
606  }
607  }
608 
609  /* Mark the state to be not halted
610  */
611  if (InstancePtr->HasMm2S) {
612  XAxiDma_BdRing *TxRingPtr;
613  TxRingPtr = XAxiDma_GetTxRing(InstancePtr);
614 
615  if (XAxiDma_HasSg(InstancePtr)) {
616  Status = XAxiDma_BdRingStart(TxRingPtr);
617  if (Status != XST_SUCCESS) {
618  xdbg_printf(XDBG_DEBUG_ERROR, "Resume: failed"
619  " to start tx ring %d\r\n", Status);
620 
621  return XST_DMA_ERROR;
622  }
623  }
624 
625  TxRingPtr->RunState = AXIDMA_CHANNEL_NOT_HALTED;
626  }
627 
628  if (InstancePtr->HasS2Mm) {
629  int RingIndex = 0;
630  for (RingIndex = 0 ; RingIndex < InstancePtr->RxNumChannels;
631  RingIndex++) {
632  XAxiDma_BdRing *RxRingPtr;
633  RxRingPtr = XAxiDma_GetRxIndexRing(InstancePtr, RingIndex);
634 
635  if (XAxiDma_HasSg(InstancePtr)) {
636  Status = XAxiDma_BdRingStart(RxRingPtr);
637  if (Status != XST_SUCCESS) {
638  xdbg_printf(XDBG_DEBUG_ERROR, "Resume: failed"
639  "to start rx ring %d\r\n", Status);
640 
641  return XST_DMA_ERROR;
642  }
643  }
644 
645  RxRingPtr->RunState = AXIDMA_CHANNEL_NOT_HALTED;
646  }
647  }
648 
649  return XST_SUCCESS;
650 }

831 u32 XAxiDma_SimpleTransfer(XAxiDma *InstancePtr, UINTPTR BuffAddr, u32 Length,
832  int Direction)
833 {
834  printf("\r\n--- Entering XAxiDma_SimpleTransfer() --- \r\n");
835 
836  u32 WordBits;
837  int RingIndex = 0;
838 
839  /* If Scatter Gather is included then, cannot submit
840  */
841  if (XAxiDma_HasSg(InstancePtr)) {
842  xdbg_printf(XDBG_DEBUG_ERROR, "Simple DMA mode is not"
843  " supported\r\n");
844 
845  return XST_FAILURE;
846  }
847 
848  if (Direction == XAXIDMA_DMA_TO_DEVICE) {
849  if ((Length < 1) ||
850  (Length > InstancePtr->TxBdRing.MaxTransferLen)) {
851  printf("XAxiDma_SimpleTransfer DMA_TO_DEVICE failed -> Exceeds Max Transfer Length\r\n");
852  return XST_INVALID_PARAM;
853  }
854 
855  if (!InstancePtr->HasMm2S) {
856  xdbg_printf(XDBG_DEBUG_ERROR, "MM2S channel is not"
857  "supported\r\n");
858  printf("XAxiDma_SimpleTransfer DMA_TO_DEVICE failed -> No MM2S.\r\n");
859  return XST_FAILURE;
860  }
861 
862  /* If the engine is doing transfer, cannot submit
863  */
864 
865  if (!(XAxiDma_ReadReg(InstancePtr->TxBdRing.ChanBase,
866  XAXIDMA_SR_OFFSET) & XAXIDMA_HALTED_MASK)) {
867  if (XAxiDma_Busy(InstancePtr, Direction)) {
868  xdbg_printf(XDBG_DEBUG_ERROR,
869  "Engine is busy\r\n");
870  return XST_FAILURE;
871  }
872  }
873 
874  if (!InstancePtr->MicroDmaMode) {
875  WordBits = (u32)((InstancePtr->TxBdRing.DataWidth) - 1);
876  } else {
877  WordBits = XAXIDMA_MICROMODE_MIN_BUF_ALIGN;
878  }
879 
880  if ((BuffAddr & WordBits)) {
881 
882  if (!InstancePtr->TxBdRing.HasDRE) {
883  xdbg_printf(XDBG_DEBUG_ERROR,
884  "Unaligned transfer without"
885  " DRE %x\r\n", (unsigned int)BuffAddr);
886 
887  return XST_INVALID_PARAM;
888  }
889  }
890 
891 
892  XAxiDma_WriteReg(InstancePtr->TxBdRing.ChanBase,
893  XAXIDMA_SRCADDR_OFFSET, LOWER_32_BITS(BuffAddr));
894  if (InstancePtr->AddrWidth > 32)
895  XAxiDma_WriteReg(InstancePtr->TxBdRing.ChanBase,
896  XAXIDMA_SRCADDR_MSB_OFFSET,
897  UPPER_32_BITS(BuffAddr));
898 
899  XAxiDma_WriteReg(InstancePtr->TxBdRing.ChanBase,
900  XAXIDMA_CR_OFFSET,
901  XAxiDma_ReadReg(
902  InstancePtr->TxBdRing.ChanBase,
903  XAXIDMA_CR_OFFSET) | XAXIDMA_CR_RUNSTOP_MASK);
904 
905  /* Writing to the BTT register starts the transfer
906  */
907  XAxiDma_WriteReg(InstancePtr->TxBdRing.ChanBase,
908  XAXIDMA_BUFFLEN_OFFSET, Length);
909  } else if (Direction == XAXIDMA_DEVICE_TO_DMA) {
910  if ((Length < 1) ||
911  (Length >
912  InstancePtr->RxBdRing[RingIndex].MaxTransferLen)) {
913  printf("XAxiDma_SimpleTransfer DEVICE_TO_DMA failed -> Exceeds Max Transfer Length.\r\n");
914  return XST_INVALID_PARAM;
915  }
916 
917 
918  if (!InstancePtr->HasS2Mm) {
919  xdbg_printf(XDBG_DEBUG_ERROR, "S2MM channel is not"
920  " supported\r\n");
921  printf("XAxiDma_SimpleTransfer DEVICE_TO_DMA failed -> No S2MM.\r\n");
922  return XST_FAILURE;
923  }
924 
925  if (!(XAxiDma_ReadReg(InstancePtr->RxBdRing[RingIndex].ChanBase,
926  XAXIDMA_SR_OFFSET) & XAXIDMA_HALTED_MASK)) {
