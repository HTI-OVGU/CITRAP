323 /*****************************************************************************/
333 static INLINE __attribute__((always_inline)) u32 Xil_EndianSwap32(u32 Data)
334 {
335  u16 LoWord;
336  u16 HiWord;
337 
338  /* get each of the half words from the 32 bit word */
339 
340  LoWord = (u16) (Data & 0x0000FFFFU);
341  HiWord = (u16) ((Data & 0xFFFF0000U) >> 16U);
342 
343  /* byte swap each of the 16 bit half words */
344 
345  LoWord = (u16)(((LoWord & 0xFF00U) >> 8U) | ((LoWord & 0x00FFU) << 8U));
346  HiWord = (u16)(((HiWord & 0xFF00U) >> 8U) | ((HiWord & 0x00FFU) << 8U));
347 
348  /* swap the half words before returning the value */
349 
350  return ((((u32)LoWord) << (u32)16U) | (u32)HiWord);
351 }
