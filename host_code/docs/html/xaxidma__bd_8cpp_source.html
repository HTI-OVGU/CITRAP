103 u32 XAxiDma_BdSetBufAddr(XAxiDma_Bd *BdPtr, UINTPTR Addr)

107

114  " %x\r\n", Addr, HasDRE, (WordLen - 1),
115  Addr & (WordLen - 1));
116 
117  return XST_INVALID_PARAM;
118  }
119  }
120 
121 #if defined(__aarch64__) || defined(__arch64__)
122  XAxiDma_BdWrite64(BdPtr, XAXIDMA_BD_BUFA_OFFSET, Addr);

263 void XAxiDma_BdSetCtrl(XAxiDma_Bd *BdPtr, u32 Data)
264 {
265  u32 RegValue = XAxiDma_BdRead(BdPtr, XAXIDMA_BD_CTRL_LEN_OFFSET);
266 
267  RegValue &= ~XAXIDMA_BD_CTRL_ALL_MASK;
268

272
